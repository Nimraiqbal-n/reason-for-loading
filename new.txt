        public ActionResult NewAward(string id)
        {
            if (Session["UID"] == null || string.IsNullOrEmpty(Session["UID"].ToString()))
            {
                return RedirectToAction("Login", "Account");
            }

            var courseInfo = GetCourseInfo(id);
            if (courseInfo == null)
            {
                ViewBag.ErrorMessage = "Course information not found.";
                return View();
            }

            var theoryData = GetCourseData(id, "Theory");
            var labData = GetCourseData(id, "Lab");

            var finalDataSet = CombineCourseData(theoryData, labData, courseInfo);

            var assessmentInfo = GetAssessmentInfo(id);
            var gradingRules = GetGradingRules(id);
            var additionalInfo = GetAdditionalCourseInfo(id);

            ViewBag.Assessment = assessmentInfo;
            ViewBag.FinalData = finalDataSet;
            ViewBag.GradeRules = gradingRules;
            ViewBag.CourseInfo = additionalInfo;

            return View();
        }

        private DataTable GetCourseInfo(string courseId)
        {
            var query = "SELECT t, L FROM CourseinfoAll WHERE id = @CourseId";
            return _dal.ExecuteQuery(query, new SqlParameter("@CourseId", courseId)).Tables[0];
        }

        private DataSet GetCourseData(string courseId, string type)
        {
            var query = "EXEC PROCFinalResultgradeNew @semid = @CourseId, @TL = @Type";
            return _dal.ExecuteQuery(query,

                new SqlParameter("@Type", type));
        }

        private DataTable GetAssessmentInfo(string courseId)
        { 
            var query = "SELECT type, TotalMarks, CASE WHEN status = 0 THEN 'Not Submitted' ELSE 'Submitted' END AS Status " +
                        "FROM Semestercourses_weightage INNER JOIN SemesterCourse_WeightageDetail " +
                        "ON SemesterCourse_WeightageDetail.SemesterCourseWeightage_ID = Semestercourses_weightage.ID " +
                        "WHERE SemCoursesInfoID = @CourseId ORDER BY Semestercourses_weightage.id";

            return _dal.ExecuteQuery(query, new SqlParameter("@CourseId", courseId)).Tables[0];
        }

        private DataTable GetGradingRules(string courseId)
        {
            var query = "SELECT * FROM Semester_Grade_Rules WHERE SemCrsID = @CourseId";
            return _dal.ExecuteQuery(query, new SqlParameter("@CourseId", courseId)).Tables[0];
        }

        private DataTable GetAdditionalCourseInfo(string courseId)
        {
            var query = "SELECT DISTINCT Semester_Course_Relative.*, scheme, coursecode, CourseName, CreditHrs, EmpName, " +
                        "Student_Course_Enrollment_count.STD_COUNT FROM (SELECT MAX(ID) AS ID, SemCourseInfoID FROM Semester_Course_Relative GROUP BY SemCourseInfoID) AS t " +
                        "INNER JOIN Semester_Course_Relative ON Semester_Course_Relative.ID = t.ID " +
                        "RIGHT OUTER JOIN Student_Course_Enrollment_count ON Student_Course_Enrollment_count.SemCourseID = Semester_Course_Relative.SemCourseInfoID " +
                        "WHERE Student_Course_Enrollment_count.SemCourseID = @CourseId";

            return _dal.ExecuteQuery(query, new SqlParameter("@CourseId", courseId)).Tables[0];
        }

        public ActionResult Index()
        {
            if (Session["semsessionno"] == null)
            {
                return RedirectToAction("Login", "Account");
            }

            var enrollmentCount = GetEnrollmentCount(Session["semsessionno"].ToString());
            var courseCount = GetCourseCount(Session["semsessionno"].ToString());
            var resultHistory = GetResultHistory(Session["semsessionno"].ToString());

            ViewBag.Enrollment = enrollmentCount;
            ViewBag.CourseCount = courseCount;
            ViewBag.ResultHistory = resultHistory;

            return View();
        }

        private int GetEnrollmentCount(string sessionId)
        {
            var query = "SELECT COUNT(DISTINCT StdRollNo) FROM Student_Course_Enrollment WHERE SemesterSessionID = @SessionId";
            var result = _dal.ExecuteQuery(query, new SqlParameter("@SessionId", sessionId));
            return Convert.ToInt32(result.Tables[0].Rows[0][0]);
        }

        private int GetCourseCount(string sessionId)
        {
            var query = "SELECT COUNT(*) FROM CourseinfoAll WHERE SemesterSessionID = @SessionId";
            var result = _dal.ExecuteQuery(query, new SqlParameter("@SessionId", sessionId));
            return Convert.ToInt32(result.Tables[0].Rows[0][0]);
        }

        private DataTable GetResultHistory(string sessionId)
        {
            var query = "SELECT coursesubmissionstatus.Status FROM CourseinfoAll " +
                        "INNER JOIN SemesterCoursesInfo ON SemesterCoursesInfo.id = CourseinfoAll.id " +
                        "INNER JOIN DegreeBatchInfo ON DegreeBatchInfo.id = SemesterCoursesInfo.DegreeBatchID " +
                        "INNER JOIN DegreeInfo ON DegreeInfo.id = DegreeBatchInfo.DegreeID " +
                        "INNER JOIN Departments ON Departments.id = DegreeInfo.DepartmentID " +
                        "INNER JOIN Faculty ON Faculty.ID = Departments.Faculty " +
                        "LEFT OUTER JOIN coursesubmissionstatus ON coursesubmissionstatus.id = CourseinfoAll.id " +
                        "WHERE CourseinfoAll.SemesterSessionID = @SessionId";

            return _dal.ExecuteQuery(query, new SqlParameter("@SessionId", sessionId)).Tables[0];
        }

        public string[,] Double_Array_store(DataSet ds)
        {
            string[,] final;
            int r = ds.Tables[0].Rows.Count;
            int c = ds.Tables[0].Columns.Count;
            if (r == 0)
            {
                r = 1;
                final = new string[r, c];
                for (int i = 0; i < r; i++)
                {
                    for (int j = 0; j < c; j++)
                    {
                        final[i, j] = null;
                    }
                }
            }
            else
            {
                final = new string[r, c];
                for (int i = 0; i < r; i++)
                {
                    for (int j = 0; j < c; j++)
                    {
                        final[i, j] = ds.Tables[0].Rows[i][j].ToString();
                    }
                }
            }
            return final;
        }
        public string[,] Array_store(DataTable dt)
        {
            int c = dt.Columns.Count;
            int r = dt.Rows.Count;
            string[,] final = new string[r + 1, c];
            for (int i = 0; i < c; i++)
            {
                final[0, i] = dt.Columns[i].ToString();
                for (int j = 0; j < r; j++)
                {
                    final[j + 1, i] = dt.Rows[j][i].ToString();
                }
            }
            return final;
        }
